    <xsl:template match="a[substring(@href,1,1)='/']" mode="html" priority="1">
      <a>
        <xsl:attribute name="href"><xsl:value-of select="$root"/><xsl:value-of select="@href"/></xsl:attribute>
        <xsl:value-of select="."/>
        <xsl:apply-templates select="./*"/>
      </a>
    </xsl:template>
